118def test_delaunay_neighbors_square(): 

119 """Tests delaunay_neighbors with 4 points forming a square.""" 

120 # For a square, Delaunay typically gives the 4 outer edges and one diagonal. 

121 # The choice of diagonal can depend on slight perturbations or library implementation. 

122 # We'll test for a common outcome. 

123 coords = np.array([[0, 0], [1, 0], [1, 1], [0, 1]]) 

124 neighbors = delaunay_neighbors(coords) 

125 assert len(neighbors) == 4 

126 

127 # Check general properties: sorted, no self-loops 

128 for i in range(len(coords)): 

129 assert neighbors[i] == sorted(neighbors[i]), f"Neighbors for node {i} not sorted" 

130 assert i not in neighbors[i], f"Node {i} is its own neighbor" 

131 for neighbor_node in neighbors[i]: 

132 assert i in neighbors[neighbor_node], f"Asymmetric neighborhood: {i} -> {neighbor_node} but not vice-versa" 

133 

134 # Expected connections for a square (0-1-2-3-0) 

135 # Node 0: connected to 1, 3. One diagonal (e.g., to 2) 

136 # Node 1: connected to 0, 2. One diagonal (e.g., to 3) 

137 # Node 2: connected to 1, 3. One diagonal (e.g., to 0) 

138 # Node 3: connected to 0, 2. One diagonal (e.g., to 1) 

139 # The exact diagonal depends on the triangulation. 

140 # Let's check that each node has degree 3 (2 square edges + 1 diagonal) 

141 # or degree 2 if it's a very flat quadrilateral that only forms 2 triangles along one diagonal. 

142 # For a unit square, it's typically 2 triangles, so most nodes have degree 3. 

143 # Example: diagonal (0, 2) -> neighbors[0]=[1, 2, 3], neighbors[1]=[0, 2], neighbors[2]=[0, 1, 3], neighbors[3]=[0, 2] 

144 # Example: diagonal (1, 3) -> neighbors[0]=[1, 3], neighbors[1]=[0, 2, 3], neighbors[2]=[1, 3], neighbors[3]=[0, 1, 2] 

145 

146 # A robust check: ensure all perimeter edges exist 

147 assert 1 in neighbors[0] and 0 in neighbors[1] # Edge 0-1 

148 assert 2 in neighbors[1] and 1 in neighbors[2] # Edge 1-2 

149 assert 3 in neighbors[2] and 2 in neighbors[3] # Edge 2-3 

150 assert 0 in neighbors[3] and 3 in neighbors[0] # Edge 3-0 

151 

152 # Check that the sum of degrees is 2 * num_edges. 

153 # For 4 points, Delaunay gives 2*N-2-k triangles (k=num hull edges), so 2*4-2-4 = 2 triangles. 

154 # Number of edges in 2 triangles = 3 edges/triangle * 2 triangles - shared_edges. 

155 # If 2 triangles share 1 edge (the diagonal), total edges = 3+3-1 = 5. 

156 # Sum of degrees = 2 * 5 = 10. 

157 # If 2 triangles share 0 edges (not possible for convex hull), total edges = 6. Sum of degrees = 12. 

158 # For a convex quad, it's 5 edges. 

159 total_degree = sum(len(n_list) for n_list in neighbors) 

160 assert total_degree == 2 * 5 # 5 edges in a typical Delaunay of a convex quadrilateral
